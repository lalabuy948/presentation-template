Project structure:

.
├── app
│   └── Main.hs           <- start of your program
├── ChangeLog.md          <- pffff
├── LICENSE               <- pfff
├── my-cool-project.cabal <- Autogenerated cabal file (do not toouch)
├── package.yaml          <- Your project setup file
├── README.md             <- pffff
├── Setup.hs              <- Stack file
├── src
│   └── Lib.hs            <- Place for your source
├── stack.yaml            <- Package settings and dependencies
└── test
    └── Spec.hs           <- pffff

routes :: ScottyM ()

  get "/allSolarObjects" $ do
    json allSolarObjects

Create an actual server

# Main.hs


main = do
    putStrLn "Starting Server on localhost:8080 ..."
    scotty 8080 routes

matchesId :: Int -> SolarObject -> Bool

get "/solarObject/:id" $ do
    id <- param "id"
